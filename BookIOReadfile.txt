public class ReadBookFromFile {
	public static void main(String[] args) throws IOException, ClassNotFoundException {
		ArrayList<Object> mylist =null;

		InputStream fin= new FileInputStream("D:\\Core_Java\\Book.bingo");

		ObjectInputStream oin = new ObjectInputStream(fin);

		try {
			while(true) {
				Object list=oin.readObject();
				if(list==null)   break;
				else {
					System.out.println(list);
				}
			}
			oin.close();
		}
		catch(Exception ex) {
			System.out.println(" ");
		}
		oin.close();
	}
}
